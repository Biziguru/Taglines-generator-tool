  <script>
    function showTaglines(tag) {
      var taglines = {
        college: [
          "Leash Your Intellectual Curiosity",
          "Pursue Knowledge and Excellence",
          "Shaping Global Minds"
        ],
        university: [
          "Unlock the Future's Potential",
          "Empowering Tomorrow's Leaders",
          "Innovation in Education"
        ],
        brand: [
          "Create Lasting Impressions",
          "Stand Out from the Crowd",
          "Your Brand, Your Identity"
        ],
        more: [
          "Innovation Beyond Boundaries",
          "Transforming Industries",
          "Driving Change for a Better Future"
        ]
      };

      var taglinesContainer = document.getElementById('taglines-container');
      var taglineElements = '';

      taglines[tag].forEach(function (tagline) {
        taglineElements += `
          <div class="tagline">
            <span class="tagline-text">${tagline}</span>
            <button class="copy-button" onclick="copyToClipboard(this)">Copy</button>
            <span class="copied-message">Copied!</span>
          </div>
        `;
      });

      taglinesContainer.innerHTML = taglineElements;
    }

    function copyToClipboard(button) {
      var taglineText = button.previousElementSibling;
      var copiedMessage = button.nextElementSibling;

      var textarea = document.createElement('textarea');
      textarea.value = taglineText.textContent;
      document.body.appendChild(textarea);
      textarea.select();
      document.execCommand('copy');
      document.body.removeChild(textarea);

      button.style.display = 'none';
      copiedMessage.style.display = 'inline-block';

      setTimeout(function () {
        copiedMessage.style.display = 'none';
        button.style.display = 'inline-block';
      }, 2000);
    }
  </script>
